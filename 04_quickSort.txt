//quick sort
#include<iostream>
using namespace std;

void swap(int arr[],int i,int j)
{
    int temp=arr[j];
    arr[j]=arr[i];
    arr[i]=temp;
}
int partition(int arr[],int l,int r)
{
    int pivot=arr[r];
    int i=l-1;
    
    for(int j=l;j<r;j++)
    {
        if(arr[j]<pivot)
        {
            i++;
            swap(arr,i,j);
        }
    }
    swap(arr,i+1,r);
    return i+1;
}
void quickSort(int arr[],int l,int r)
{
    int pi=partition(arr,l,r);
    
    quickSort(arr,l,pi-1);
    quickSort(arr,pi+1,r);
    
}
int main()
{
    int arr[]={4,3,2,6,1,7};
    int length=sizeof(arr)/sizeof(arr[0]);
    
    cout<<"Original array: ";
    for(int i=0;i<length;i++)
    {
        cout<<arr[i]<<" ";
    }
    
    cout<<endl;
    //fucntion call
    quickSort(arr,0,5);
    
    cout<<"Sorted array: ";
    for(int i=0;i<length;i++)
    {
        cout<<arr[i]<<" ";
    }
}
