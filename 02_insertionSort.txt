//insertion sort
#include<iostream>
using namespace std;

//function to sort the array
void insertionSort(int arr[],int length)
{
  for(int i=1;i<length;i++)
  {
      int current=arr[i];
      int j=i-1;
      
      while(arr[j]>current &&j>=0)
      {
          arr[j+1]=arr[j];
          j--;
      }
      arr[j+1]=current;
  }
}
int main()
{
    int arr[]={4,2,6,3,8,1};
    int length=sizeof(arr)/sizeof(arr[0]);
    
    cout<<"Original array:"<<" ";
    for(int i=0;i<length;i++)
    {
        cout<<arr[i]<<" ";
    }
    
    cout<<endl;
    //function call
    insertionSort(arr,length);
    
    //printing sorted array
    cout<<"Sorted array: ";
    for(int i=0;i<length;i++)
    {
        cout<<arr[i]<<" ";
    }
}S
